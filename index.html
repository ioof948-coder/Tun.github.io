import csv from datetime import datetime FILENAME = "expense_tracker.csv" def initialize_file(): """สร้างไฟล์ CSV พร้อมส่วนหัว (Header) ถ้ายังไม่มีไฟล์อยู่""" try: with open(FILENAME, mode='x', newline='', encoding='utf-8') as file: writer = csv.writer(file) writer.writerow(["วันที่", "ประเภท", "รายการ", "จำนวนเงิน (บาท)"]) except FileExistsError: pass def add_transaction(transaction_type): """ฟังก์ชันเพิ่มรายการรายรับ หรือ รายจ่าย""" print(f"\n--- เพิ่ม{transaction_type} ---") detail = input("ระบุรายละเอียดรายการ: ").strip() while True: try: amount = float(input("ระบุจำนวนเงิน (บาท): ")) if amount <= 0: print("จำนวนเงินต้องมากกว่า 0 บาท") continue break except ValueError: print("กรุณากรอกตัวเลขที่ถูกต้อง") date_now = datetime.now().strftime("%Y-%m-%d %H:%M") with open(FILENAME, mode='a', newline='', encoding='utf-8') as file: writer = csv.writer(file) writer.writerow([date_now, transaction_type, detail, amount]) print(f"บันทึก {transaction_type}: {detail} จำนวน {amount} บาท เรียบร้อยแล้ว!") def view_summary(): """ฟังก์ชันแสดงรายการทั้งหมดและสรุปยอดคงเหลือ""" print("\n--- ประวัติการเงินและสรุปยอด ---") total_income = 0 total_expense = 0 try: with open(FILENAME, mode='r', encoding='utf-8') as file: reader = csv.reader(file) header = next(reader) # ข้ามหัวข้อ print(f"{'วันที่':<18} | {'ประเภท':<6} | {'รายการ':<20} | {'จำนวนเงิน':<10}") print("-" * 65) for row in reader: date, t_type, detail, amount = row amount = float(amount) if t_type == "รายรับ": total_income += amount elif t_type == "รายจ่าย": total_expense += amount print(f"{date:<18} | {t_type:<6} | {detail:<20} | {amount:<10,.2f}") print("-" * 65) print(f"รวมรายรับ: {total_income:,.2f} บาท") print(f"รวมรายจ่าย: {total_expense:,.2f} บาท") print(f"ยอดเงินคงเหลือสุทธิ: {total_income - total_expense:,.2f} บาท") except FileNotFoundError: print("ยังไม่มีข้อมูลการเงินในระบบ เริ่มบันทึกรายการก่อนนะ!") def main(): """ฟังก์ชันหลักควบคุมเมนู""" initialize_file() while True: print("\n===== โปรแกรมบันทึกรายรับ-รายจ่าย =====") print("1. บันทึกรายรับ") print("2. บันทึกรายจ่าย") print("3. ดูประวัติและสรุปยอดเงิน") print("4. ออกจากโปรแกรม") choice = input("เลือกเมนู (1-4): ").strip() if choice == '1': add_transaction("รายรับ") elif choice == '2': add_transaction("รายจ่าย") elif choice == '3': view_summary() elif choice == '4': print("ขอบคุณที่ใช้งานโปรแกรม ออมเงินวันละนิดจิตแจ่มใสครับ! 👋") break else: print("เลือกเมนูไม่ถูกต้อง กรุณาเลือก 1-4 เท่านั้น") if __name__ == "__main__": main()
